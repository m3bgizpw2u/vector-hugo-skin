{{/* References list — the Wikipedia-style footer block. Pairs with `{{< ref "key" >}}` markers in the article body. Renders the `[^key]: text` footnote definitions discovered in the article body (in first-cite document order) using the same markup MediaWiki emits for `
    `: a `
    ` with a "References" heading (carrying the matching `[edit]` affordance), followed by a numbered list whose first character is an up-arrow backlink. Usage: {{< references >}} Mark-up per entry:
  1. ↑ {citation text, markdownified}
    2. No front-matter requirement. The keys and their texts come from the Markdown footnote block at the bottom of the article body — `[^key]: text` — the same syntax the Markdown Guide describes. Goldmark's own `
    ` block is suppressed via `[markup.goldmark.extensions.footnote] enable = false` in `hugo.toml`, so this output is the only references block the page emits. */}} {{ $page := .Page }} {{ $order := $page.Scratch.Get "refFootnoteOrder" | default (slice) }} {{ $defs := $page.Scratch.Get "refFootnoteDefs" }} {{ if not $defs }} {{ $raw := $page.RawContent }} {{ $lines := split $raw "\n" }} {{ $defs = dict }} {{ range $lines }} {{ $line := trim . " \r" }} {{ $matches := findRE `^\[\^([A-Za-z0-9_-]+)\]:[[:space:]]+(.+)$` $line }} {{ if $matches }} {{ $keyName := replaceRE `^\[\^([A-Za-z0-9_-]+)\]:.*` "$1" (index $matches 0) }} {{ $body := replaceRE `^\[\^[A-Za-z0-9_-]+\]:[[:space:]]+` "" (index $matches 0) }} {{ $defs = merge $defs (dict $keyName $body) }} {{ end }} {{ end }} {{ $page.Scratch.Set "refFootnoteDefs" $defs }} {{ end }}

    References [edit]
